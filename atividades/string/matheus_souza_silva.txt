#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main() {
	char nome_completo[30];
	char nome[30];
	char sobrenome[30];
	int i = 0;
	
	printf("--------------------------------\n");
	printf("      Don't Worry Airlines      \n");
	printf("--------------------------------\n");
	
	printf("Insira o seu Nome Completo: ");
	scanf("%[^\n]s",&nome_completo);
	
    	char *primeiro_espaco = strchr(nome_completo, ' ');
    	if(primeiro_espaco != NULL) {
        	int tam = primeiro_espaco - nome_completo;
        	strncpy(nome, nome_completo, tam);
        	nome[tam] = '\0';
    	} else {
        	strcpy(nome, nome_completo);
    	}
    
    	nome[0] = toupper(nome[0]);
    	for(i = 1; i < strlen(nome); i++){
    		nome[i] = tolower(nome[i]);
	}
    
    	char *ultimo_espaco = strrchr(nome_completo, ' ');
    	if(ultimo_espaco != NULL) {
		strcpy(sobrenome, ultimo_espaco + 1);
    	}
    
	for(i = 0; i < strlen(sobrenome); i++){
		sobrenome[i] = toupper(sobrenome[i]);
	}
	
	printf("+-------------------------+\n");
	printf("|       %s/%s             |\n",sobrenome,nome);
	printf("|       Boa Viagem!       |\n");
	printf("+-------------------------+");

	return 0;
}
